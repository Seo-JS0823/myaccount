CATEGORYS TABLE

CREATE TABLE CATEGORYS(
CATEGORY_ID NUMBER PRIMARY KEY,
CATE_NAME VARCHAR2(30) NOT NULL,
ACOT_FLAG VARCHAR2(30) NOT NULL CHECK(ACOT_FLAG = 'INCOME' OR ACOT_FLAG = 'SPENDING')
);

CREATE SEQUENCE CATEGORY_SEQ
START WITH 1
INCREMENT BY 1
NOCACHE;
